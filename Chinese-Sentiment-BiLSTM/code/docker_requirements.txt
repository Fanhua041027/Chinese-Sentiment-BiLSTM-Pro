# =============================================================================
# docker_requirements.txt - Docker 容器内 Python 依赖
# =============================================================================
#
# 本文件列出在 Docker 容器中运行 BERT 基线实验所需的 Python 依赖。
# 基础镜像 pytorch/pytorch:2.3.0-cuda12.1-cudnn8-runtime 已预装 PyTorch，
# 因此无需在此声明 torch / torchvision 等 PyTorch 核心包。
#
# 容器内通过 main.py 子命令运行的任务：
#   - python main.py baseline_bert      BERT 微调训练与评估
#   - python main.py plot_bert_cm       BERT 混淆矩阵绘制
#
# =============================================================================

# -----------------------------------------------------------------------------
# 核心依赖：Transformers 库
# -----------------------------------------------------------------------------
# HuggingFace Transformers：提供 bert-base-chinese 预训练模型、分词器、
# 以及 BertForSequenceClassification 等开箱即用的分类模型。
# 这是 BERT 基线实验的核心依赖。
transformers

# -----------------------------------------------------------------------------
# 数据处理
# -----------------------------------------------------------------------------
# pandas：用于读取 train.csv / test.csv 数据集，处理 DataFrame 格式数据。
pandas

# scikit-learn：提供 accuracy_score、classification_report、confusion_matrix
# 等评估指标计算函数，用于生成分类报告和混淆矩阵。
scikit-learn

# jieba：中文分词库（BERT 实际使用 BertTokenizer，但部分辅助脚本可能用到）。
jieba

# opencc-python-reimplemented：OpenCC 简繁转换库，用于将繁体中文转为简体。
opencc-python-reimplemented

# -----------------------------------------------------------------------------
# 可视化
# -----------------------------------------------------------------------------
# matplotlib：Python 绑图库，用于绘制混淆矩阵、模型对比图等。
matplotlib

# seaborn：基于 matplotlib 的统计可视化库，用于绘制热力图风格的混淆矩阵。
seaborn

# -----------------------------------------------------------------------------
# 工具库
# -----------------------------------------------------------------------------
# tqdm：进度条库，用于显示训练和评估过程的进度。
tqdm

# pandarallel：pandas 并行处理库，加速大规模数据预处理（可选，部分脚本使用）。
pandarallel
